–•–æ—Ä–æ—à–æ, –∞–≤—Ç–æ–º–∞—Ç–∏–∑–∏—Ä–æ–≤–∞—Ç—å –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞ –Ω–∞–ø—Ä—è–º—É—é —Å **Kindle Cloud Reader** ‚Äî –Ω–µ —Ç–∞–∫ –ø—Ä–æ—Å—Ç–æ –∏–∑-–∑–∞ –∑–∞—â–∏—Ç–Ω—ã—Ö –º–µ—Ö–∞–Ω–∏–∑–º–æ–≤ Amazon. –ù–æ –µ—Å—Ç—å **—Å–ø–æ—Å–æ–± –æ–±–æ–π—Ç–∏ —ç—Ç–æ —Å –ø–æ–º–æ—â—å—é –∞–≤—Ç–æ–º–∞—Ç–∏–∑–∞—Ü–∏–∏ –±—Ä–∞—É–∑–µ—Ä–∞** —á–µ—Ä–µ–∑ Python + Selenium. –°–∫—Ä–∏–ø—Ç –±—É–¥–µ—Ç:

1. –õ–æ–≥–∏–Ω–∏—Ç—å—Å—è –Ω–∞ [read.amazon.com](https://read.amazon.com/)
2. –û—Ç–∫—Ä—ã–≤–∞—Ç—å –Ω—É–∂–Ω—É—é –∫–Ω–∏–≥—É (–ø–æ —Å—Å—ã–ª–∫–µ –∏–ª–∏ –ø–æ ASIN)
3. –õ–∏—Å—Ç–∞—Ç—å —Å—Ç—Ä–∞–Ω–∏—Ü—ã
4. –ò–∑–≤–ª–µ–∫–∞—Ç—å —Ç–µ–∫—Å—Ç —Å –∫–∞–∂–¥–æ–π —Å—Ç—Ä–∞–Ω–∏—Ü—ã
5. –°–æ—Ö—Ä–∞–Ω—è—Ç—å –≤—Å—ë –≤ –æ–¥–∏–Ω `.txt` —Ñ–∞–π–ª

### üß† –°–∫—Ä–∏–ø—Ç –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–≥–æ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è —Ç–µ–∫—Å—Ç–∞ –∏–∑ Kindle Cloud Reader

```python
import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from webdriver_manager.chrome import ChromeDriverManager

# –ù–∞—Å—Ç—Ä–æ–π–∫–∏
AMAZON_EMAIL = "your_email@example.com"
AMAZON_PASSWORD = "your_password"
BOOK_URL = "https://read.amazon.com/?asin=B009SE1Z9E"
OUTPUT_FILE = "kindle_book.txt"
PAGES_TO_READ = 50  # –°–∫–æ–ª—å–∫–æ —Å—Ç—Ä–∞–Ω–∏—Ü –ø—Ä–æ–ª–∏—Å—Ç–∞—Ç—å

# –ó–∞–ø—É—Å–∫ –±—Ä–∞—É–∑–µ—Ä–∞
driver = webdriver.Chrome(ChromeDriverManager().install())
driver.get("https://read.amazon.com")

# –í—Ö–æ–¥ –≤ –∞–∫–∫–∞—É–Ω—Ç Amazon
time.sleep(2)
driver.find_element(By.ID, "ap_email").send_keys(AMAZON_EMAIL)
driver.find_element(By.ID, "continue").click()
time.sleep(2)
driver.find_element(By.ID, "ap_password").send_keys(AMAZON_PASSWORD)
driver.find_element(By.ID, "signInSubmit").click()

# –û—Ç–∫—Ä—ã–≤–∞–µ–º –∫–Ω–∏–≥—É
time.sleep(8)
driver.get(BOOK_URL)
time.sleep(10)

# –ö–ª–∏–∫ –ø–æ —Ü–µ–Ω—Ç—Ä—É, —á—Ç–æ–±—ã —É–±—Ä–∞—Ç—å –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å
driver.find_element(By.TAG_NAME, "body").click()
time.sleep(2)

# –û—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª –ø–æ —Å—Ç—Ä–∞–Ω–∏—Ü–∞–º
with open(OUTPUT_FILE, 'w', encoding='utf-8') as f:
    for page in range(PAGES_TO_READ):
        try:
            # –ù–∞–π—Ç–∏ —ç–ª–µ–º–µ–Ω—Ç —Ç–µ–∫—Å—Ç–∞ (–º–æ–∂–µ—Ç –∑–∞–≤–∏—Å–µ—Ç—å –æ—Ç –∫–Ω–∏–≥–∏)
            content_elements = driver.find_elements(By.CLASS_NAME, 'textLayer')
            if not content_elements:
                content_elements = driver.find_elements(By.CLASS_NAME, 'kcrPage')

            page_text = ""
            for elem in content_elements:
                page_text += elem.text + "\n"

            f.write(f"\n\n=== Page {page + 1} ===\n")
            f.write(page_text.strip())

            # –ù–∞–∂–∏–º–∞–µ–º —Å—Ç—Ä–µ–ª–∫—É "–≤–ø–µ—Ä—ë–¥"
            body = driver.find_element(By.TAG_NAME, "body")
            body.send_keys(Keys.ARROW_RIGHT)
            time.sleep(2)
        except Exception as e:
            print(f"–û—à–∏–±–∫–∞ –Ω–∞ —Å—Ç—Ä–∞–Ω–∏—Ü–µ {page+1}: {e}")
            break

print(f"–°–æ—Ö—Ä–∞–Ω–µ–Ω–æ {PAGES_TO_READ} —Å—Ç—Ä–∞–Ω–∏—Ü –≤ —Ñ–∞–π–ª: {OUTPUT_FILE}")
driver.quit()
```
